\newlength{\indenthang}
\newenvironment{hanging}[1]{
\setlength{\indenthang}{#1}
\begin{list}{xx}{ \itemindent=0pt
\labelwidth=\indenthang
\setlength{\leftmargin}{\labelwidth}
\addtolength{\leftmargin}{4pt}
}}{
\end{list}
}
